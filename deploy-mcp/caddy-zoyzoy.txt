# ZoyZoy Hub vhosts — append to /etc/caddy/Caddyfile (the same file serving
# skoolai.cloud + mcp.skoolai.cloud). Reuses the existing (common) snippet.
# After saving:  caddy validate --config /etc/caddy/Caddyfile && systemctl reload caddy
#
# Each block points at a service bound to 127.0.0.1 by docker-compose.yml.
# DNS: add A records (or a wildcard *.skoolai.cloud) -> 167.86.100.217 for every
# hostname below, or Caddy can't issue its cert.

# CRM admin dashboard + same-origin API  (basic-auth protected)
# /api/* -> api container, everything else -> CRM. The CRM bundle calls
# https://admin.skoolai.cloud/api/* (same origin) so the browser auto-sends creds.
# Replace BCRYPT_HASH_HERE with: caddy hash-password --plaintext '<password>'
admin.skoolai.cloud {
	import common
	basic_auth {
		admin BCRYPT_HASH_HERE
	}
	@api path /api/*
	reverse_proxy @api 127.0.0.1:4000
	reverse_proxy 127.0.0.1:3000
}

# Backend API direct host  (basic-auth protected; sites use internal api:4000)
api.skoolai.cloud {
	import common
	basic_auth {
		admin BCRYPT_HASH_HERE
	}
	reverse_proxy 127.0.0.1:4000
}
